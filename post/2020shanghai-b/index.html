2020shanghai B - 缓冲区结算、括号嵌套的递归

sicp
2020-8-6 11:13

题面:https://ac.nowcoder.com/acm/contest/6871/B

分析

典型的字符串处理，而且出题人强调了没有括号嵌套（真亚撒西嘤嘤嘤）。

因为分子式不同部分的计算方式是不一样的，可以用缓冲的思想来解决这个问题。

在不考虑括号嵌套的情况下

#include <iostream>
#include "string"
using namespace std;
typedef long long ll;
string a;
int main(){
    cin>>a;
    ll ans=0,cnt=0,res=0,sum=0;
    int len=a.length();
    for(int i=0;i<a.length();)
    {
        if(a[i]=='C'||a[i]=='H'||a[i]=='O'){
             if(a[i]=='C')cnt=13;
              if(a[i]=='H')cnt=1;
            if(a[i]=='O')cnt=17;
            i++;
            res=0;
            while(i<len&&a[i]>='0'&&a[i]<='9'){
                res=res*10+a[i]-'0';
                i++;
            }
            if(!res)ans+=cnt;
            else ans+=cnt*res;
        }

        //遇到左括号，就先把未加进最终结果的非括号内元素们算入结果，清空缓冲区，准备计算括号内的分子式
        if(a[i]=='('&&i<len){
          sum+=ans;
            ans=0;
            i++;
        }
        //遇到右括号，结算括号内部的
        if(a[i]==')'&&i<len){
            i++;
            res=0;
             while(i<len&&a[i]>='0'&&a[i]<='9'){
                res=res*10+a[i]-'0';
                i++;
            }
            if(!res)sum+=ans;
            else sum+=ans*res;
            ans=0;
        }
    }
    sum+=ans;
    printf("%lld",sum);
}

用递归的方式处理括号嵌套

如果要考虑括号内部的嵌套，就需要用到递归的方法了。

先说缓冲区的部分，遇到常规字符C,H,O，将缓冲区算入结果，将常规字符的值读入缓冲区，以处理后面可能会出现的数字，括号等情况。

遇到左括号，也是先将缓冲区算入结果，然后进入下一层函数，计算括号内部的分子式，遇到右括号时内部分子式返回来。如果内部还有括号，同理，会进入再下一层函数。最后将括号内的值读入缓冲区。

遇到数字，就会继续往下读，直到把整个数字段都读出来（这个过程会转换为具体的数字times）。然后将缓冲区内的值乘以times算入结果，清空缓冲区（这里清空缓冲区是必要的，虽然读入下一个数据的时候会覆盖它，但是如果已经是最后一个数据，就会跳出循环，执行最后的缓冲区内数据算入结果，如果不清空会多算一次）。

#include <cstdio>
#include "string"
#define ll long long
using namespace std;
char in[1000050];
int len;
ll cal(int &now){
    ll tmp=0,sum=0,times=0;
    for(now;now<len;){
        if(in[now]>='0'&&in[now]<='9'){
            times=(in[now]-'0');
            now++;
            while(in[now]>='0'&&in[now]<='9'){
                times*=10;
                times+=(in[now]-'0');
                now++;
            }
            sum+=tmp*times;
            tmp=0;
            continue;
        }
        //遇到左括号时，先将缓冲区内部加入结果，然后进入下一层函数，计算内部分子式，遇到右括号再返回来
        if(in[now]=='('){
            sum+=tmp;
            now++;
            tmp=cal(now);
        }
        else if(in[now]==')'){
            sum+=tmp;
            now++;
            return sum;
        }
        //遇到常规字符也是把上一个缓冲区的内容加入结果，把当前的加入缓冲区，以处理后面可能会出现的数字
        if(in[now]=='C'){
            sum+=tmp;
            tmp=13;
            now++;
        }else if(in[now]=='H'){
            sum+=tmp;
            tmp=1;
            now++;
        }else if(in[now]=='O'){
            sum+=tmp;
            tmp=17;
            now++;
        }
    }
    sum+=tmp;
    return sum;
}
void solve(int T){
    scanf("%s",in);
    len=strlen(in);
    int now=0;
    printf("%lld",cal(now));
}
signed main(){
    int t=1;
    //scanf("%d",&t);
    for(int i=1;i<=t;i++){
        solve(i);
    }
}
